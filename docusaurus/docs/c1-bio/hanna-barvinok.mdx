## –í–Ω–µ—Å–æ–∫

### –ê–Ω–∞–ª—ñ–∑ –∫–ª—é—á–æ–≤–∏—Ö —Ç–≤–æ—Ä—ñ–≤

1.  **"–ü'—è–Ω–∏—Ü—è"**: –¶–µ–π —Ç–≤—ñ—Ä ‚Äî —Å–ø—Ä–∞–≤–∂–Ω—è –µ–Ω—Ü–∏–∫–ª–æ–ø–µ–¥—ñ—è —Å–µ–ª—è–Ω—Å—å–∫–æ–≥–æ –≥–æ—Ä—è. –ì–æ–ª–æ–≤–Ω–∞ –≥–µ—Ä–æ—ó–Ω—è —Å—Ç—Ä–∞–∂–¥–∞—î –≤—ñ–¥ –ø–∏—è—Ü—Ç–≤–∞ —á–æ–ª–æ–≤—ñ–∫–∞, —è–∫–∏–π –ø–æ—Å—Ç—É–ø–æ–≤–æ –¥–µ–≥—Ä–∞–¥—É—î –≤—ñ–¥ –∑–∞–º–æ–∂–Ω–æ–≥–æ –≥–æ—Å–ø–æ–¥–∞—Ä—è –¥–æ –∂–µ–±—Ä–∞–∫–∞. –ë–∞—Ä–≤—ñ–Ω–æ–∫ –¥–µ—Ç–∞–ª—å–Ω–æ, –º–∞–π–∂–µ –∫–ª—ñ–Ω—ñ—á–Ω–æ –æ–ø–∏—Å—É—î –ø—Ä–æ—Ü–µ—Å —Ä–æ–∑–ø–∞–¥—É –æ—Å–æ–±–∏—Å—Ç–æ—Å—Ç—ñ —á–µ—Ä–µ–∑ –∞–ª–∫–æ–≥–æ–ª—å, –∞–ª–µ —Ñ–æ–∫—É—Å —Ä–æ–±–∏—Ç—å –Ω–∞ —Å—Ç—Ä–∞–∂–¥–∞–Ω–Ω—è—Ö –¥—Ä—É–∂–∏–Ω–∏. –ê–≤—Ç–æ—Ä–∫–∞ –Ω–µ –∑–∞—Å—É–¥–∂—É—î —á–æ–ª–æ–≤—ñ–∫–∞ –ø—Ä—è–º–æ, –≤–æ–Ω–∞ –ø–æ–∫–∞–∑—É—î —Ç—Ä–∞–≥–µ–¥—ñ—é –∑–∞–ª–µ–∂–Ω–æ—Å—Ç—ñ, —è–∫–∞ —Ä—É–π–Ω—É—î —Ü—ñ–ª–∏–π —Ä—ñ–¥. –ì–µ—Ä–æ—ó–Ω—è –Ω–µ –∫–∏–¥–∞—î —á–æ–ª–æ–≤—ñ–∫–∞, –∞ –Ω–∞–º–∞–≥–∞—î—Ç—å—Å—è –π–æ–≥–æ –≤—Ä—è—Ç—É–≤–∞—Ç–∏ –º–æ–ª–∏—Ç–≤–æ—é —ñ —Ç–µ—Ä–ø—ñ–Ω–Ω—è–º, —â–æ –≤—ñ–¥–æ–±—Ä–∞–∂–∞—î —Ç–∏–ø–æ–≤—É –º–æ–¥–µ–ª—å –ø–æ–≤–µ–¥—ñ–Ω–∫–∏ —Ç–æ–≥–æ —á–∞—Å—É.
2.  **"–•–∞—Ç–Ω—î –ª–∏—Ö–æ" (1861)**: –¶–µ–π —Ç–≤—ñ—Ä ‚Äî —Å–ø—Ä–∞–≤–∂–Ω—è –ø—Å–∏—Ö–æ–ª–æ–≥—ñ—á–Ω–∞ –¥—Ä–∞–º–∞, —è–∫–∞ –ø–µ—Ä–µ–≥—É–∫—É—î—Ç—å—Å—è –∑ "–ö–∞–π–¥–∞—à–µ–≤–æ—é —Å—ñ–º'—î—é" –ù–µ—á—É—è-–õ–µ–≤–∏—Ü—å–∫–æ–≥–æ, –∞–ª–µ –Ω–∞–ø–∏—Å–∞–Ω–∞ –∑–Ω–∞—á–Ω–æ —Ä–∞–Ω—ñ—à–µ. –Ø–∫—â–æ –ù–µ—á—É–π-–õ–µ–≤–∏—Ü—å–∫–∏–π –∑–æ–±—Ä–∞–∂—É—î –ø–æ–±—É—Ç–æ–≤–µ –ø–µ–∫–ª–æ —á–∞—Å—Ç–æ —á–µ—Ä–µ–∑ –ø—Ä–∏–∑–º—É –≥—É–º–æ—Ä—É —Ç–∞ –≥—Ä–æ—Ç–µ—Å–∫—É, —Ç–æ –ë–∞—Ä–≤—ñ–Ω–æ–∫ —Ä–æ–∑–∫—Ä–∏–≤–∞—î –π–æ–≥–æ —è–∫ –≥–ª–∏–±–æ–∫—É, –±–µ–∑–≤–∏—Ö—ñ–¥–Ω—É —Ç—Ä–∞–≥–µ–¥—ñ—é.

    –ê–≤—Ç–æ—Ä–∫–∞ –¥–µ—Ç–∞–ª—å–Ω–æ –∞–Ω–∞–ª—ñ–∑—É—î –º–µ—Ö–∞–Ω—ñ–∫—É —Ä–æ–¥–∏–Ω–Ω–æ—ó —Ç–∏—Ä–∞–Ω—ñ—ó. –°–≤–µ–∫—Ä—É—Ö–∞ —Ç—É—Ç –≤–∏—Å—Ç—É–ø–∞—î –Ω–µ –ø—Ä–æ—Å—Ç–æ —è–∫ "–∑–ª–∞ –±–∞–±–∞", –∞ —è–∫ –Ω–æ—Å—ñ–π –ø–∞—Ç—Ä—ñ–∞—Ä—Ö–∞–ª—å–Ω–æ—ó –≤–ª–∞–¥–∏, —è–∫–∞ —Å–∞–º–∞ –∫–æ–ª–∏—Å—å –±—É–ª–∞ –∂–µ—Ä—Ç–≤–æ—é —ñ —Ç–µ–ø–µ—Ä –≤—ñ–¥—Ç–≤–æ—Ä—é—î —Ü–µ–π —Ü–∏–∫–ª –Ω–∞—Å–∏–ª—å—Å—Ç–≤–∞ –Ω–∞ –Ω–æ–≤—ñ–π –∂–µ—Ä—Ç–≤—ñ. –ì–∞–Ω–Ω–∞ –ë–∞—Ä–≤—ñ–Ω–æ–∫, –º–æ–∂–ª–∏–≤–æ, —ñ–Ω—Ç—É—ó—Ç–∏–≤–Ω–æ, –ø–µ—Ä—à–æ—é –≤ —É–∫—Ä–∞—ó–Ω—Å—å–∫—ñ–π –ª—ñ—Ç–µ—Ä–∞—Ç—É—Ä—ñ —Ç–æ—Ä–∫–Ω—É–ª–∞—Å—è —Ç–µ–º–∏ —Ü–∏–∫–ª–∏—á–Ω–æ—Å—Ç—ñ –¥–æ–º–∞—à–Ω—å–æ–≥–æ –Ω–∞—Å–∏–ª–ª—è.
    –î–æ—Å–∏—Ç—å —Ü—ñ–∫–∞–≤–∏–º —î —ñ —Ñ—ñ–Ω–∞–ª —Ç–≤–æ—Ä—É. –ù–∞ –≤—ñ–¥–º—ñ–Ω—É –≤—ñ–¥ —Ä–æ–º–∞–Ω—Ç–∏—á–Ω–∏—Ö –±–∞–ª–∞–¥, –¥–µ –≥–µ—Ä–æ—ó–Ω—è –ø–æ–º–∏—Ä–∞—î –∞–±–æ —Ç—ñ–∫–∞—î, —É "–•–∞—Ç–Ω—å–æ–º—É –ª–∏—Ö—É" –≥–µ—Ä–æ—ó–Ω—è _–∑–º–∏—Ä—è—î—Ç—å—Å—è_, –∞–ª–µ —Ü–µ —Å–º–∏—Ä–µ–Ω–Ω—è ‚Äî —Ü–µ —Å–º–µ—Ä—Ç—å –¥—É—à—ñ –ø—Ä–∏ –∂–∏–≤–æ–º—É —Ç—ñ–ª—ñ. –¢–∞–∫–∏–π —Ä–µ–∞–ª—ñ–∑–º –±—É–≤ —à–æ–∫—É—é—á–∏–º –¥–ª—è —á–∏—Ç–∞—á—ñ–≤, –∑–≤–∏–∫–ª–∏–º –¥–æ —Å–µ–Ω—Ç–∏–º–µ–Ω—Ç–∞–ª—å–Ω–∏—Ö —Ä–æ–∑–≤'—è–∑–æ–∫.

3.  **"–í—ñ—Ä–Ω–∞ –ø–∞—Ä–∞"**: –û–¥–∏–Ω –∑ –Ω–µ–±–∞–≥–∞—Ç—å–æ—Ö —Ç–≤–æ—Ä—ñ–≤ –∑ –≤—ñ–¥–Ω–æ—Å–Ω–æ —â–∞—Å–ª–∏–≤–∏–º —Ñ—ñ–Ω–∞–ª–æ–º, –¥–µ –æ—Å–ø—ñ–≤—É—î—Ç—å—Å—è —ñ–¥–µ–∞–ª –≤—ñ—Ä–Ω–æ–≥–æ –∫–æ—Ö–∞–Ω–Ω—è, —è–∫–µ –¥–æ–ª–∞—î –≤—Å—ñ –ø–µ—Ä–µ—à–∫–æ–¥–∏. –¢—É—Ç –ë–∞—Ä–≤—ñ–Ω–æ–∫ –≤–∏—Å—Ç—É–ø–∞—î —è–∫ —Ä–æ–º–∞–Ω—Ç–∏–∫, —É—Ç–≤–µ—Ä–¥–∂—É—é—á–∏ —Å–∏–ª—É –ø–æ—á—É—Ç—Ç—è. –ê–ª–µ –π —Ç—É—Ç –≤–æ–Ω–∞ –Ω–µ —É–Ω–∏–∫–∞—î —Ä–µ–∞–ª—ñ–∑–º—É: —â–∞—Å—Ç—è –≥–µ—Ä–æ—ó–≤ –∑–¥–æ–±—É—Ç–µ —á–µ—Ä–µ–∑ –≤–∞–∂–∫—ñ –≤–∏–ø—Ä–æ–±—É–≤–∞–Ω–Ω—è —ñ —Ä–æ–∑–ª—É–∫—É, –≤–æ–Ω–æ –Ω–µ –¥–∞—î—Ç—å—Å—è –∑–∞–¥–∞—Ä–º–∞.

### –ú–æ–≤–Ω–∞ –ø–∞–ª—ñ—Ç—Ä–∞ —ñ –¥—ñ–∞–ª–µ–∫—Ç–∏–∑–º–∏

–û–∫—Ä–µ–º–æ—ó —É–≤–∞–≥–∏ –∑–∞—Å–ª—É–≥–æ–≤—É—î –º–æ–≤–∞ –ì–∞–Ω–Ω–∏ –ë–∞—Ä–≤—ñ–Ω–æ–∫. –í–æ–Ω–∞ –ø–∏—Å–∞–ª–∞ —á–µ—Ä–Ω—ñ–≥—ñ–≤—Å—å–∫–∏–º –¥—ñ–∞–ª–µ–∫—Ç–æ–º, –ø–æ–≤–Ω–∏–º –∞—Ä—Ö–∞—ó–∑–º—ñ–≤ —Ç–∞ —Å–ø–µ—Ü–∏—Ñ—ñ—á–Ω–∏—Ö –∑–≤–æ—Ä–æ—Ç—ñ–≤, —è–∫—ñ —Å—å–æ–≥–æ–¥–Ω—ñ –º–æ–∂—É—Ç—å –∑–¥–∞—Ç–∏—Å—è –Ω–µ–∑–≤–∏—á–Ω–∏–º–∏, –∞–ª–µ —Å–∞–º–µ –≤–æ–Ω–∏ –Ω–∞–¥–∞—é—Ç—å —ó—ó –ø—Ä–æ–∑—ñ –Ω–µ–ø–æ–≤—Ç–æ—Ä–Ω–æ–≥–æ —Å–º–∞–∫—É.
–í–æ–Ω–∞ —á–∞—Å—Ç–æ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É–≤–∞–ª–∞ —Å–ª–æ–≤–∞, —è–∫—ñ –ø–æ–∑–Ω–∞—á–∞—é—Ç—å —Ç–æ–Ω–∫—ñ –≤—ñ–¥—Ç—ñ–Ω–∫–∏ –ø–æ—á—É—Ç—Ç—ñ–≤ —Ç–∞ —Å—Ç–∞–Ω—ñ–≤: "–Ω—É–¥–∏—Ç–∏ —Å–≤—ñ—Ç–æ–º" (–±—É—Ç–∏ –≤ –¥–µ–ø—Ä–µ—Å—ñ—ó), "–ø–æ–±–∏–≤–∞—Ç–∏—Å—è" (—Å–∏–ª—å–Ω–æ –≥–æ—Ä—é–≤–∞—Ç–∏), "—Ç—É–∂–∏—Ç–∏", "–º–ª—ñ—Ç–∏". –á—ó –ª–µ–∫—Å–∏–∫–æ–Ω –±–∞–≥–∞—Ç–∏–π –Ω–∞ "–∂—ñ–Ω–æ—á—ñ" —Å–ª–æ–≤–∞ ‚Äî –Ω–∞–∑–≤–∏ —Ç–∫–∞–Ω–∏–Ω, –ø—Ä–∏–∫—Ä–∞—Å, —Å—Ç—Ä–∞–≤, –ø—Ä–æ—Ü–µ—Å—ñ–≤ –≤–∏—à–∏–≤–∞–Ω–Ω—è —Ç–∞ —Ç–∫–∞—Ü—Ç–≤–∞. –¶–µ —Ä–æ–±–∏—Ç—å —ó—ó —Ç–µ–∫—Å—Ç–∏ –µ—Ç–Ω–æ–≥—Ä–∞—Ñ—ñ—á–Ω–æ —Ü—ñ–Ω–Ω–∏–º–∏ –¥–∂–µ—Ä–µ–ª–∞–º–∏. –í–æ–Ω–∞ –Ω–µ –≤–∏–≥–∞–¥—É–≤–∞–ª–∞ "–ª—ñ—Ç–µ—Ä–∞—Ç—É—Ä–Ω—É" –º–æ–≤—É, –∞ —Ñ—ñ–∫—Å—É–≤–∞–ª–∞ —Ç—É, —è–∫–æ—é –≥–æ–≤–æ—Ä–∏–ª–∞ —Å–∞–º–∞ —ñ —ó—ó –æ—Ç–æ—á–µ–Ω–Ω—è.
–ù–∞–ø—Ä–∏–∫–ª–∞–¥, –∑–∞–º—ñ—Å—Ç—å —Å—Ç–∞–Ω–¥–∞—Ä—Ç–Ω–æ–≥–æ "—Å—É–º—É–≤–∞—Ç–∏", –≤–æ–Ω–∞ –º–æ–≥–ª–∞ –≤–∂–∏—Ç–∏ "–≤'—è–Ω—É—Ç–∏ —Å–µ—Ä—Ü–µ–º", —â–æ —Å—Ç–≤–æ—Ä—é—î –∑–æ–≤—Å—ñ–º —ñ–Ω—à–∏–π, –±—ñ–ª—å—à –ø–æ–µ—Ç–∏—á–Ω–∏–π –æ–±—Ä–∞–∑.

### –°—Ç–∏–ª—ñ—Å—Ç–∏—á–Ω—ñ –æ—Å–æ–±–ª–∏–≤–æ—Å—Ç—ñ: –°–µ–Ω—Ç–∏–º–µ–Ω—Ç–∞–ª—å–Ω–∏–π —Ä–µ–∞–ª—ñ–∑–º

–á—ó —Å—Ç–∏–ª—å –Ω–∞–∑–≤–∞–ª–∏ **—Å–µ–Ω—Ç–∏–º–µ–Ω—Ç–∞–ª—å–Ω–∏–º —Ä–µ–∞–ª—ñ–∑–º–æ–º**. –¶–µ –ø–æ—î–¥–Ω–∞–Ω–Ω—è —Ç–æ—á–Ω–æ—Å—Ç—ñ –ø–æ–±—É—Ç–æ–≤–∏—Ö –æ–ø–∏—Å—ñ–≤ –∑ –≥–ª–∏–±–æ–∫–æ—é —É–≤–∞–≥–æ—é –¥–æ –ø–æ—á—É—Ç—Ç—ñ–≤.

- **–ï—Ç–Ω–æ–≥—Ä–∞—Ñ—ñ—á–Ω—ñ—Å—Ç—å:** –í–æ–Ω–∞ –æ–ø–∏—Å—É—î –æ–¥—è–≥ (–æ—á—ñ–ø–∫–∏, –ø–ª–∞—Ö—Ç–∏), —ó–∂—É, —ñ–Ω—Ç–µ—Ä'—î—Ä —Ö–∞—Ç–∏ –∑ —Ç–æ—á–Ω—ñ—Å—Ç—é —Ñ–æ—Ç–æ–≥—Ä–∞—Ñ–∞.
- **–ü—Å–∏—Ö–æ–ª–æ–≥—ñ–∑–º:** –í–æ–Ω–∞ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É—î –≤–Ω—É—Ç—Ä—ñ—à–Ω—ñ –º–æ–Ω–æ–ª–æ–≥–∏, —Å–Ω–∏, —Å–Ω–∏-–≤—ñ–∑—ñ—ó, —â–æ–± –ø–µ—Ä–µ–¥–∞—Ç–∏ —Å—Ç–∞–Ω –¥—É—à—ñ –≥–µ—Ä–æ—ó–Ω—ñ.
- **–ú–æ–≤–∞:** –á—ó –º–æ–≤–∞ ‚Äî —Ü–µ –∂–∏–≤–∞, —Å–æ–∫–æ–≤–∏—Ç–∞ –Ω–∞—Ä–æ–¥–Ω–∞ –≥–æ–≤—ñ—Ä–∫–∞ –ß–µ—Ä–Ω—ñ–≥—ñ–≤—â–∏–Ω–∏, –Ω–µ –∑—ñ–ø—Å–æ–≤–∞–Ω–∞ –∫–Ω–∏–∂–Ω–∏–º–∏ –≤–ø–ª–∏–≤–∞–º–∏. –í–æ–Ω–∞ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É—î —Ä—ñ–¥–∫—ñ—Å–Ω—ñ —Å–ª–æ–≤–∞ ("–≥–æ—Ä—é–≤–∞—Ç–∏", "–Ω—É–¥–∏—Ç–∏ —Å–≤—ñ—Ç–æ–º", "–ø–æ–±–∏–≤–∞—Ç–∏—Å—è"), —è–∫—ñ —Å—Ç–≤–æ—Ä—é—é—Ç—å –æ—Å–æ–±–ª–∏–≤—É –º–µ–ª–æ–¥–∏–∫—É —Ç–µ–∫—Å—Ç—É.

### –ß–æ–ª–æ–≤—ñ—á—ñ –æ–±—Ä–∞–∑–∏ —É —Ç–≤–æ—Ä—á–æ—Å—Ç—ñ –ë–∞—Ä–≤—ñ–Ω–æ–∫

–•–æ—á–∞ –ì–∞–Ω–Ω–∞ –ë–∞—Ä–≤—ñ–Ω–æ–∫ –≤—ñ–¥–æ–º–∞ —è–∫ —Å–ø—ñ–≤–µ—Ü—å –∂—ñ–Ω–æ—á–æ—ó –¥–æ–ª—ñ, —ó—ó —á–æ–ª–æ–≤—ñ—á—ñ –ø–µ—Ä—Å–æ–Ω–∞–∂—ñ –∑–∞—Å–ª—É–≥–æ–≤—É—é—Ç—å –Ω–∞ –æ–∫—Ä–µ–º—É —É–≤–∞–≥—É. –£ —ó—ó –ø—Ä–æ–∑—ñ —á–æ–ª–æ–≤—ñ–∫–∏ —á–∞—Å—Ç–æ –ø–æ—Å—Ç–∞—é—Ç—å —É –¥–≤–æ—Ö —ñ–ø–æ—Å—Ç–∞—Å—è—Ö: –∞–±–æ —è–∫ –¥–∂–µ—Ä–µ–ª–æ —Å—Ç—Ä–∞–∂–¥–∞–Ω–Ω—è (–ø–∏—è–∫–∏, —Ç–∏—Ä–∞–Ω–∏, –±–∞–π–¥—É–∂—ñ –≥–æ—Å–ø–æ–¥–∞—Ä—ñ), –∞–±–æ —è–∫ —ñ–¥–µ–∞–ª—ñ–∑–æ–≤–∞–Ω—ñ, –º–∞–π–∂–µ —Å–≤—è—Ç—ñ —Ñ—ñ–≥—É—Ä–∏, —â–æ –∑'—è–≤–ª—è—é—Ç—å—Å—è –ª–∏—à–µ –µ–ø—ñ–∑–æ–¥–∏—á–Ω–æ.
–£ –ø–æ–≤—ñ—Å—Ç—ñ "–í—ñ—Ä–Ω–∞ –ø–∞—Ä–∞" –≤–æ–Ω–∞ –º–∞–ª—é—î –æ–±—Ä–∞–∑ —á–æ–ª–æ–≤—ñ–∫–∞, –∑–¥–∞—Ç–Ω–æ–≥–æ –Ω–∞ –≤—ñ—á–Ω—É –≤—ñ–¥–¥–∞–Ω—ñ—Å—Ç—å, —â–æ –±—É–ª–æ —Ä—ñ–¥–∫—ñ—Å—Ç—é –¥–ª—è —Ç–æ–≥–æ—á–∞—Å–Ω–æ—ó –ª—ñ—Ç–µ—Ä–∞—Ç—É—Ä–∏, —è–∫–∞ —á–∞—Å—Ç–æ –∑–æ–±—Ä–∞–∂–∞–ª–∞ —Å–µ–ª—è–Ω–∏–Ω–∞ –≥—Ä—É–±–∏–º —ñ –ø—Ä–∏–º—ñ—Ç–∏–≤–Ω–∏–º. –ë–∞—Ä–≤—ñ–Ω–æ–∫ –∂–µ –Ω–∞–¥—ñ–ª—è—î —Å–≤–æ—ó—Ö –≥–µ—Ä–æ—ó–≤ (—ñ —á–æ–ª–æ–≤—ñ–∫—ñ–≤, —ñ –∂—ñ–Ω–æ–∫) —Ç–æ–Ω–∫–æ—é –¥—É—à–µ–≤–Ω–æ—é –æ—Ä–≥–∞–Ω—ñ–∑–∞—Ü—ñ—î—é. –í–æ–Ω–∞ —Å—Ç–≤–µ—Ä–¥–∂—É—î, —â–æ "–ø—ñ–¥ —Å—ñ—Ä—è–∫–æ–º" –±'—î—Ç—å—Å—è —Ç–∞–∫–µ –∂ —á—É—Ç–ª–∏–≤–µ —Å–µ—Ä—Ü–µ, —è–∫ —ñ –ø—ñ–¥ "—Ñ—Ä–∞–∫–æ–º". –¶–µ –±—É–ª–æ –Ω–æ–≤–∞—Ç–æ—Ä—Å—Ç–≤–æ–º: —Å—Ç–≤–µ—Ä–¥–∂—É–≤–∞—Ç–∏, —â–æ –µ–º–æ—Ü—ñ–π–Ω–∏–π —ñ–Ω—Ç–µ–ª–µ–∫—Ç –Ω–µ –∑–∞–ª–µ–∂–∏—Ç—å –≤—ñ–¥ —Å–æ—Ü—ñ–∞–ª—å–Ω–æ–≥–æ —Å—Ç–∞—Ç—É—Å—É.
