import LevelLanding from '@site/src/components/LevelLanding';
import { A1_UNITS, A1_NON_MODULE_SLUGS } from '@site/src/data/a1-modules';

{/*
  Dynamic deployment detection.
  `import.meta.glob` enumerates every `.mdx` sibling under `a1/` at
  build time. Whatever exists on disk is treated as `done` (clickable);
  everything else stays `locked`. No hand-edited status list — when a
  new module's MDX lands, the landing page reflects it automatically.
*/}

export const _deployedSlugs = new Set(
  Object.keys(import.meta.glob('./*.mdx', { eager: false }))
    .map((p) => p.replace(/^\.\//, '').replace(/\.mdx$/, ''))
    .filter((slug) => !A1_NON_MODULE_SLUGS.has(slug))
);

export const _modulesWithStatus = A1_UNITS.map((group) => ({
  unit: group.unit,
  items: group.items.map((m) => ({
    ...m,
    status: _deployedSlugs.has(m.slug) ? 'done' : 'locked',
  })),
}));

<LevelLanding
  client:load
  level="A1"
  title="Перші кроки · First Steps"
  subtitle="Ukrainian for absolute beginners — 55 modules, ~1,200 target words"
  moduleCount={55}
  wordTarget={1200}
  color="#2E7D32"
  modules={_modulesWithStatus}
/>
