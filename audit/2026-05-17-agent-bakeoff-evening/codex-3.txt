def find_russianisms(text: str) -> list[tuple[str, str]]:
    """Detect selected Ukrainian Russianisms and return suggested replacements."""
    patterns = (
        ("на протязі", "протягом"),
        ("учбовий", "навчальний"),
        ("співпадати", "збігатися"),
        ("мішати", "заважати"),
        ("приймати участь", "брати участь"),
    )

    lowered = text.casefold()
    matches: list[tuple[int, int, str]] = []

    for phrase, suggestion in patterns:
        start = 0
        needle = phrase.casefold()
        while True:
            index = lowered.find(needle, start)
            if index == -1:
                break
            matches.append((index, len(phrase), suggestion))
            start = index + 1

    matches.sort(key=lambda match: match[0])
    return [(text[index:index + length], suggestion) for index, length, suggestion in matches]
